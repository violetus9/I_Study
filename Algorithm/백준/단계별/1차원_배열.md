  const fs = require('fs');
  const input = fs.readFileSync('/dev/stdin').toString().split('\n');

  let mul = 1;
  for(let i = 0; i < 3; i++){
      mul *= parseInt(input[i]);
  }
  let stringMul = String(mul);

  let cnt = Array(10).fill(0);
  for(let i = 0; i < 10; i++){
      cnt[parseInt(stringMul[i])] += 1;
  }
  cnt.forEach(function(i){
      console.log(i);    
  });
